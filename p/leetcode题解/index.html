671. 二叉树中第二小的节点

class Solution {
public:
    int ans = -1;
    int root_val;
    void dfs(TreeNode* root){
        if(root==nullptr) return ;
        if(ans!=-1 && root->val>ans) return ;
        if(root->val>root_val) ans = root->val;
        dfs(root->left);
        dfs(root->right);
    }
    int findSecondMinimumValue(TreeNode* root) {
        root_val = root->val;
        dfs(root);
        return ans;
    }
};

863. 二叉树中所有距离为 K 的结点

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*,TreeNode*> parent;
    vector<int> ans;
    void findParent(TreeNode* root){
        if(root==NULL) return ;
        if(root->left){
            parent[root->left]=root;
            findParent(root->left);
        }
        if(root->right){
            parent[root->right]=root;
            findParent(root->right);
        }
    }

    void dfs(TreeNode* root,int depth,int k,TreeNode* preNode){
        if(root==NULL) return ;
        if(depth==k) {
            ans.push_back(root->val);
            return ;
        }
        if(root->left && root->left!=preNode){
            dfs(root->left,depth+1,k,root);
        }
        if(root->right && root->right!=preNode){
            dfs(root->right,depth+1,k,root);
        }
        if(parent[root] && parent[root]!=preNode){
            dfs(parent[root],depth+1,k,root);
        }
        
    }
    vector<int> distanceK(TreeNode* root, TreeNode* target, int k) {
        findParent(root);
        dfs(target,0,k,NULL);
        return ans;
    }
};

581. 最短无序连续子数组

class Solution {
public:
    int findUnsortedSubarray(vector<int>& nums) {
        if(is_sorted(nums)) return 0;
        int n = nums.size();
        vector<int> origin(nums);
        sort(nums.begin(),nums.end());
        int left = 0;
        while(left < n && origin[left]==nums[left]) left++;
        int right = n-1;
        while(right >= 0 && origin[right]==nums[right]) right--;
        return right - left + 1;
    }
    bool is_sorted(vector<int> &v) {
        for(int i=0;i<v.size()-1;i++){
            if(v[i]>v[i+1]) return false ;
        }
        return true;
    }

};

611. 有效三角形的个数

class Solution {
public:
    int triangleNumber(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int n = nums.size();
        int ans = 0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                // nums[i] + nums[j] > nums[k]
                int l = j+1;
                int r = n-1;
                while(l<r){
                    int mid = l + r +1 >> 1;
                    if(nums[i] + nums[j] > nums[mid]) {
                        l = mid;
                    }else{
                        r = mid - 1;
                    }
                }
                //cout<< l << endl;
                if(l < n && nums[i] + nums[j] > nums[l]){
                    ans += l-j;
                }
            }
        }
        return ans;
    }
};

802. 找到最终的安全状态

class Solution {
public:
    vector<vector<int>> g;
    vector<int> v;
    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        g = graph;
        int n = graph.size();
        v = vector<int>(n,0);
        vector<int> ans;
        for(int i=0;i<n;i++){
            if(safe(i)) ans.push_back(i);
        }
        return ans;
        
    }
    bool safe(int x){
        if(v[x] > 0){
            //已经访问过一次
            return v[x] == 2; //是否是终点
        }
        v[x] = 1; //访问一次
        for(int next : g[x]){
            if(!safe(next)) return false;
        }
        v[x] = 2;
        return true;
    }
    
};

1583. 统计不开心的朋友

class Solution {
public:
    int unhappyFriends(int n, vector<vector<int>>& preferences, vector<vector<int>>& pairs) {
        int ans = 0;
        vector<vector<int>>order(n,vector<int>(n));
        for(int i=0;i<n;i++) {
            for(int j=0;j<n-1;j++) {
                order[i][preferences[i][j]] = j;
            }
        }
        vector<int> match(n);
        for(auto pr : pairs) {
            match[pr[0]] = pr[1];
            match[pr[1]] = pr[0];
        }
        for(int x=0;x<n;x++){
            int y = match[x];
            int index = order[x][y];
            for(int i=0;i<index;i++){
                int u = preferences[x][i];
                int v = match[u];
                if(order[u][x]<order[u][v]){
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};

526. 优美的排列

class Solution {
public:
    vector<vector<int>> match;
    vector<bool> visit;
    int ans = 0;

    void dfs(int index,int n){
        if(index == n+1) {
            ans++;
            return ;
        }

        for(auto &x : match[index]){
            if(!visit[x]){
                visit[x] = true;
                dfs(index+1,n);
                visit[x] = false;
            }
            
        }
    }

    int countArrangement(int n) {
        match.resize(n+1);
        visit.resize(n+1);
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                if(i%j == 0 || j%i == 0){
                    match[i].push_back(j);
                }
            }
        }
        dfs(1,n);
        return ans;

    }
};
